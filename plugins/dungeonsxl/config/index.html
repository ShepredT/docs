    配置文件（DungeonsXL）

    #版本
configVersion: 
#选择语言文件 对应目录是在plugins/DungeonsXL/languages
language: english 
#是否调用经济插件
enableEconomy: false
groupAdaptersEnabled: false
chatEnabled: true
chatFormat:
  edit: '&2[Edit] &r%player_name%: '
  game: '&2[Game] %group_color%%player_name%: &r'
  group: '&2%group_color%[%group_name%] %player_name%: &r'
  spy: '&2[Chat Spy] %player_name%: &r'
#教程选项
tutorial:
  #是否启用
  activated: false
  #教程副本名称
  dungeon: tutorial
  #玩家进入后所在用户组[default]
  startGroup: default
  #完成后玩家进入所在的组[player]
  endgroup: player
#颜色标识
groupColorPriority:
- DARK_BLUE
- LIGHT_RED
- YELLOW
- LIGHT_GREEN
- PURPLE
- ORANGE
- BLACK
- LIGHT_BLUE
- DARK_GREEN
- DARK_RED
- LIGHT_GRAY
- CYAN
- MAGENTA
- DARK_GRAY
- PINK
#公告间隔
announcementInterval: 30.0
#是否发送下一关Title
sendFloorTitle: true
globalDeathMessagesDisabled: true
#DungeonsXL支持任何自定义怪物插件
#您可以在此处注册占位符标记和spawn命令
#自定义外部怪物插件的应该使用怪物[Mob]标志和M触发器
#由自定义外部怪物生成插件的怪物应该会被kill击杀计数器识别 (/dxl game)
externalMobProviders: {}
    # providerTag: spawn 命令
    # "%%" 为占位符，即placeholders插件
    XY: custommobplugin spawnmob %mob% %world% %x% %y% %z%
#资源包
resourcePacks: {}
#多少世界使用一个副本
maxInstances: 10
editInstanceRemovalDelay: 5
strictMovementCheckEnabled: true
#安全模式
secureMode:
  #是否启用
  enabled: false
  #是否允许玩家在编辑模式下打开背包
  openInventories: false
  #是否允许玩家在编辑模式下放置物品 （还是丢弃扶贫）
  dropItems: false
  #每5秒检查玩家
  #检查玩家是否在编辑副本
  #如果发现错误该玩家会被送回spawn
  checkInterval: 5.0
backupMode: ON_DISABLE_AND_SAVE
lobbyContainersEnabled: false
default: {}
